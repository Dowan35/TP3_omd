1   package fr.esir.omd.ci;

3   import java.io.FileOutputStream;
4   import java.io.FileWriter;
5   import java.io.IOException;
6   import java.util.List;
7   import org.slf4j.LoggerFactory;
8   import org.slf4j.Logger;
9   
10  import org.apache.commons.csv.CSVFormat;
11  import org.apache.commons.csv.CSVPrinter;
12  
13  import com.google.gson.Gson;
14  import com.google.gson.GsonBuilder;
15  import com.itextpdf.text.Document;
16  import com.itextpdf.text.DocumentException;
17  import com.itextpdf.text.Paragraph;
18  import com.itextpdf.text.pdf.PdfWriter;
19  
20  public class FileHandler {
21      private static final Logger loger = LoggerFactory.getLogger(TaskManager.class);
22  
23      public void saveAsJson(List<Task> tasks, String filePath) throws IOException {
24          Gson gson = new GsonBuilder().setPrettyPrinting().create();
25          try (FileWriter writer = new FileWriter(filePath)) {
26              gson.toJson(tasks, writer);
27          } catch (Exception e) {
28              loger.error("Impossible de sauvegarder les taches en jason.");
29          }
30      }
31  
32      public void saveAsCsv(List<Task> tasks, String filePath) throws IOException {
33          try (CSVPrinter printer = new CSVPrinter(new FileWriter(filePath),
34                  CSVFormat.DEFAULT.builder().setHeader("Title", "Description", "Completed").build())) {
35              for (Task task : tasks) {
36                  printer.printRecord(task.getTitle(), task.getDescription(), task.isCompleted());
37              }
38          } catch (Exception e) {
39              loger.error("Impossible de sauvegarder les taches en csv.");
40          }
41      }
42  
43      public void generatePdf(List<Task> tasks, String filePath) throws DocumentException, IOException {
44          Document document = new Document();
45          try {
46              PdfWriter.getInstance(document, new FileOutputStream(filePath));
47              document.open();
48              for (Task task : tasks) {
49                  document.add(new Paragraph("Title: " + task.getTitle()));
50                  document.add(new Paragraph("Description: " + task.getDescription()));
51                  document.add(new Paragraph("Completed: " + task.isCompleted()));
52                  document.add(new Paragraph("\n"));
53              }
54          } catch (Exception e) {
55              loger.error("Impossible de sauvegarder les taches en pdf.");
56          }
57          document.close();
58      }
59  }
