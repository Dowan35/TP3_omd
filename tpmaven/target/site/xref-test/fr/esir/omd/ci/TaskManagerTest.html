14  public class TaskManagerTest {
15  
16      private TaskManager taskManager;
17  
18      @BeforeEach
19      public void setUp() {
20          taskManager = new TaskManager();
21      }
22  
23      @Test
24      public void testAddTask() {
25          // Création d'une tâche fictive avec Mockito
26          Task mockTask = mock(Task.class);
27          when(mockTask.getTitle()).thenReturn("Tâche 1");
28  
29          // Ajout de la tâche
30          taskManager.addTask(mockTask);
31  
32          // Vérifications
33          List<Task> tasks = taskManager.getTasks();
34          assertEquals(1, tasks.size(), "Le nombre de tâches doit être 1");
35          assertEquals("Tâche 1", tasks.get(0).getTitle(), "Le titre de la tâche doit correspondre");
36      }
37  
38      @Test
39      public void testGetTasksWhenEmpty() {
40          // Vérifier qu'aucune tâche n'est présente au départ
41          List<Task> tasks = taskManager.getTasks();
42          assertNotNull(tasks, "La liste des tâches ne doit pas être nulle");
43          assertTrue(tasks.isEmpty(), "La liste des tâches doit être vide");
44      }
45  
46      @Test
47      public void testAddMultipleTasks() {
48          // Création de plusieurs tâches fictives
49          Task mockTask1 = mock(Task.class);
50          Task mockTask2 = mock(Task.class);
51  
52          when(mockTask1.getTitle()).thenReturn("Tâche 1");
53          when(mockTask2.getTitle()).thenReturn("Tâche 2");
54  
55          // Ajout des tâches
56          taskManager.addTask(mockTask1);
57          taskManager.addTask(mockTask2);
58  
59          // Vérifications
60          List<Task> tasks = taskManager.getTasks();
61          assertEquals(2, tasks.size(), "Le nombre de tâches doit être 2");
62          assertEquals("Tâche 1", tasks.get(0).getTitle());
63          assertEquals("Tâche 2", tasks.get(1).getTitle());
64      }
